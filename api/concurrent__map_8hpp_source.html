328  //allocates space to fit count elements
329  void reserve(size_t count) {
330  if (count < _upper_bound.load(std::memory_order_acquire))
331  return;
332  std::lock_guard<mutex_type> l(_locks[0]);
333  if (count < _upper_bound.load(std::memory_order_acquire))
334  return;
335  auto new_size = _buckets.size() * 2;
336  while (new_size * LoadFactor / 100 < count) {
337  new_size *= 2;
338  }
339  return resize(new_size, 1);
340  }

374  void resize(size_t new_size, size_t lock_index) {
375  if (lock_index < ConcurrencyLevel) {
376  std::lock_guard<mutex_type> l(_locks[lock_index]);
377  return resize(new_size, lock_index + 1);
378  }
379  //everything is locked
380  //move old buckets
381  std::vector<Bucket, typename allocator_type::template rebind<Bucket>::other> old_buckets(std::move(_buckets));
382  bucket_flag = ((size_t) - 1) % new_size;
383  _upper_bound.store(new_size * LoadFactor / 100, std::memory_order_release);
384  _buckets.resize(new_size);
385 for (Bucket & buk : old_buckets) {
386  bool done = false;
387 for (auto & el : buk.arr) {
388  if (el.state == ElemState::UNASSIGNED) {
389  done = true;
390  break;
391  }
392  getBucket(hash_(el.key)).insertNoDuplicateCheck(std::move(el));
393  }
394  if (done)
395  continue;
396 for (auto & el : buk.overflow) {
397  getBucket(hash_(el.key)).insertNoDuplicateCheck(std::move(el));
398  }
399  }
