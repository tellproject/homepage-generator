412  void shrink_to_fit() {
413  auto s = size();
414  auto c = capacity();
415  if (s == c) return;
416  if (c <= max_in_buffer_size()) return;
417  if (s <= max_in_buffer_size()) {
418  auto ptr = get_ptr();
419  arr[0] = (unsigned char)(s);
420  std::copy(ptr, ptr + s, arr.begin() + 1);

425  std::copy(begin(), end(), nptr);

432  void reserve(size_type new_cap = 0) {
433  if (new_cap <= capacity()) return;
434  if (new_cap > max_size()) throw std::length_error("Cannot make a bigger string than max_size");
435  // we at least double the capacity
436  auto sz = size();
437  new_cap = std::max(capacity() * 2, new_cap);
438  auto nptr = alloc.allocate(new_cap + 1);
439  if (!nptr)
440  throw std::bad_alloc();

442  std::copy(ptr, ptr + size() + 1, nptr);
443  if (arr[0] != nullchar) {
444  arr[0] = nullchar;
445  set_size(sz);
446  } else {

448  }
