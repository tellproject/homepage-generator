255 template <typename Handler, typename Result>
256 void RpcResponseResult<Handler, Result>::onResponse(uint32_t messageType, crossbow::buffer_reader& message) {
257  static_assert(std::is_same<typename std::underlying_type<decltype(Handler::MessageType)>::type,
258  uint32_t>::value, "Given message type is not of the correct type");
259 
260  LOG_ASSERT(!done(), "Response is already done");
261  LOG_ASSERT(mState == RpcResponseState::UNSET, "Result is already set");
262 
263  if (messageType == std::numeric_limits<uint32_t>::max()) {
264  setError(message.read<uint64_t>(), Handler::errorCategory());
265  return;
266  }
267 
268  if (messageType != crossbow::to_underlying(Handler::MessageType)) {
269  setError(error::wrong_type);
270  return;
271  }
272 
273  static_cast<Handler*>(this)->processResponse(message);
274  if (mState == RpcResponseState::UNSET) {
275  LOG_ASSERT(!done(), "Response is already done");
276  setError(std::future_errc::broken_promise);
277  }
278 }

477 template <typename Fun, typename Message>
478 bool RpcClientSocket::sendRequest(std::shared_ptr<RpcResponse> response, Message messageType, uint32_t length,
479  Fun fun) {
480  static_assert(std::is_same<typename std::underlying_type<Message>::type, uint32_t>::value,
481  "Given message type is not of the correct type");
482  LOG_ASSERT(crossbow::to_underlying(messageType) != std::numeric_limits<uint32_t>::max(), "Invalid message type");
483 
484  mWaitingRequests.wait(response->fiber(), [this] () {
485  return (state() != ConnectionState::CONNECTING) && (mSyncResponses.size() < mMaxPendingResponses);
486  });
487 
488  if (!isConnected()) {
489  response->onAbort(std::make_error_code(std::errc::connection_aborted));
490  return false;
491  }
492 
493  ++mUserId;
494 
495  MessageId messageId(mUserId, false);
496  std::error_code ec;
497  this->writeMessage(messageId, crossbow::to_underlying(messageType), length, std::move(fun), ec);
498  if (ec) {
499  response->onAbort(std::move(ec));
500  return false;
501  }
502 
503  mSyncResponses.emplace(mUserId, std::move(response));
504  return true;
505 }
