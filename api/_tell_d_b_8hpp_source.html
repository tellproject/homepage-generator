118 private: // members
119  std::unique_ptr<tell::store::SingleTransactionRunner<telldb_context>> mTxRunner;
120  tell::store::TransactionType mTxType;
121 private: // construction
122  TransactionFiber(tell::store::ClientManager<telldb_context>& clientManager, tell::store::TransactionType txType)
123  : mTxRunner(new tell::store::SingleTransactionRunner<telldb_context>(clientManager))
124  , mTxType(txType)
125  {}
126 private: // private access
127  template<class Fun>
128  void exec(Fun fun) {
129  mTxRunner->execute([this, fun](tell::store::ClientHandle& handle, telldb_context& context) {
130  if (context.mContext.indexes == nullptr) {
131  context.mContext.setIndexes(impl::createIndexes(handle));
132  }
133  try {
134  auto clientTransaction = handle.startTransaction(mTxType);
135  Transaction transaction(handle, clientTransaction, context.mContext, mTxType);
136  context.executeHandler(fun, transaction);
137  } catch (std::exception& e) {
138  std::cerr << "Exception: " << e.what() << std::endl;
139  } catch (...) {
140  // This should never happen
141  std::cerr << "Got an unknown error" << std::endl;
142  }
143  });
144  }
145 public: // construction
146  TransactionFiber(const TransactionFiber&) = delete;
147  TransactionFiber(TransactionFiber&& other)
148  : mTxRunner(std::move(other.mTxRunner))
149  {}
150  TransactionFiber& operator=(const TransactionFiber&) = delete;
151  TransactionFiber& operator=(TransactionFiber&& other) {
152  mTxRunner = std::move(other.mTxRunner);
153  }

164  }
