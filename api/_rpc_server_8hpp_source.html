93 template <typename Manager, typename Socket>
94 void RpcServerManager<Manager, Socket>::onConnection(InfinibandSocket socket, const crossbow::string& data) {
95  LOG_TRACE("Adding connection");
96  try {
97  auto con = static_cast<Manager*>(this)->createConnection(std::move(socket), data);
98  if (con == nullptr) {
99  return;
100  }
101 
102  typename decltype(mSocketsMutex)::scoped_lock _(mSocketsMutex, false);
103  __attribute__((unused)) auto res = mSockets.insert(con);
104  LOG_ASSERT(res.second, "New connection already in connection set");
105  } catch (std::system_error& e) {
106  LOG_ERROR("Error accepting connection [error = %1% %2%]", e.code(), e.what());
107  }
108 }
