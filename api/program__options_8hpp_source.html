191  std::ostream &print_help(std::ostream &os) const {
192  if (ignore_short && ignore_long) return os;
193  os << " ";
194  if (!ignore_short) {
195  os << '-' << Name;
196  if (!ignore_long)

380 template<class O>
381 int parse(O &opts, int argc, const char** argv) {
382  int i;
383  for (i = 1; i < argc; ++i) {
384  const char* str = argv[i];
385  if (str[0] == '-' && str[1] == '-' && (std::isspace(str[2]) || str[2] == '\0')) {
386  return i + 1;
387  } else if (str[0] == '-' && str[1] == '-') {
388  // parse a long option
389  opts->parse_long(i, argv);
390  } else if (str[0] == '-') {
391  if (str[2] != '\0')
392  throw parse_error("Long options must start with '--'");
393  // parse short option
394  opts->parse_short(str[1], i, argv);
395  } else {
396  // either an error or remaining options
397  for (int j = i; j < argc; ++j) {
398  if (argv[j][0] == '-' && argv[j][1] == '-' && argv[j][2] == '\0')
399  return i;
400  if (argv[j][0] == '-')
401  throw inexpected_value(str);
402  }
403  return i;
404  }
405  }
406  return i;
407 }
