176  static void destroy() {
177  if (destroyed_) return;
178  Create::destroy(instance_);
179  instance_ = nullptr;
180  destroyed_ = true;
181  }
182 public:
183  static reference instance() {
184  static_assert(Create::supports_recreation || !lifetime_traits<LifetimePolicy>::supports_recreation,
185  "The creation policy does not support instance recreation, while the lifetime does support it.");
186  if (!instance_) {
187  std::lock_guard<Mutex> l(mutex_);
188  if (!instance_) {
189  if (destroyed_) {
190  destroyed_ = false;
191  LifetimePolicy::on_dead_ref();

193  instance_ = Create::create();

195  }
196  }
197  return *instance_;
198  }
